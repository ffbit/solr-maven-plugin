27      public void resolve(List<Artifact> artifacts) {
28          // Brute-force implementation
29          for (Artifact artifact : artifacts) {

31          }

34      public Artifact resolve(Artifact artifact) {
35          ArtifactRequest request = new ArtifactRequest();
36          request.setArtifact(artifact);
37          request.setRepositories(remoteRepositories);
38  
39          try {
40              ArtifactResult result =
41                      repositorySystem.resolveArtifact(repositorySession, request);
42              return result.getArtifact();
43          } catch (ArtifactResolutionException e) {
44              throw new RuntimeException(e.getMessage(), e);
45          }
46      }
47  
48      public Artifact resolve(String coordinates) {
49          return resolve(new DefaultArtifact(coordinates));
50      }
