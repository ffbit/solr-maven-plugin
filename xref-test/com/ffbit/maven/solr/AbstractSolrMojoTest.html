80      /**
81       * This method is a whole work around which makes it read configuration
82       * from test pom.xml file.
83       *
84       * @param session
85       * @param execution
86       * @return
87       * @throws Exception
88       */
89      @Override
90      protected Mojo lookupConfiguredMojo(MavenSession session, MojoExecution execution)
91              throws Exception {
92          MavenProject project = session.getCurrentProject();
93          MojoDescriptor mojoDescriptor = execution.getMojoDescriptor();
94  
95          Mojo mojo = (Mojo) lookup(mojoDescriptor.getRole(), mojoDescriptor.getRoleHint());
96  
97          ExpressionEvaluator evaluator = new PluginParameterExpressionEvaluator(session, execution);
98  
99          Xpp3Dom configuration = null;
100         Plugin plugin = project.getPlugin(mojoDescriptor.getPluginDescriptor().getPluginLookupKey());
101         if (plugin != null) {
102             configuration = (Xpp3Dom) plugin.getConfiguration();
103         }
104 
105         // I just swapped arguments to achieve the right behavior
106         configuration = Xpp3Dom.mergeXpp3Dom(configuration, execution.getConfiguration());
107 
108         PlexusConfiguration pluginConfiguration = new XmlPlexusConfiguration(configuration);
109 
110         // Reflection work around
111         Field configurator = AbstractMojoTestCase.class
112                 .getDeclaredField("configurator");
113         configurator.setAccessible(true);
114         Method method = ComponentConfigurator.class.getDeclaredMethod(
115                 "configureComponent", Object.class, PlexusConfiguration.class,
116                 ExpressionEvaluator.class, ClassRealm.class);
117         method.invoke(configurator.get(this), mojo, pluginConfiguration, evaluator,
118                 getContainer().getContainerRealm());
119 
120         return mojo;
121     }
